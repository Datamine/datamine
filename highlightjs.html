from random import randint

def getran():
    d1,d2,d4=randint(1,6),randint(1,6),randint(1,6)
    if d1+d2==3:
        return True
    elif d2+d4==3:
        return True
    elif d1+d4==3:
        return True
    else:
        return False

def getcount():
    count=0
    for i in range(1000000):
        if getran():
            count+=1
    print count
